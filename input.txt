!-----Example of Syntatically Correct Code-----
!Code begins with keyword 'begin'
begin
    !Declarative Statement with Multiple ID's
    int a, b, c;

    !Assignment Statement
    a = 2;
    b = 4;

    !Expression Assignment
    a = b + 2;

    !Term + Factor Assignemnt
    a = b + (a * b);

    !If - Else Statments
    if(a == b){
        a = c
    };
    else if(a < b){
        a = b
    };
    else
        a = 2;

    !While Statement
    while(a <= b){
        a = a + 1
    }


end
!'end' makes the end of the code
